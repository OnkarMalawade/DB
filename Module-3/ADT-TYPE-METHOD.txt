Create type employee_t as Object(
  	name varchar2(30),
    ssn varchar2(30),
    salary NUMBER,
    member function raise_sal 
    	return number
);

-- TYPE CREATED..
create type body employee_t as
    Member function raise_sal return Number is 
		Begin 
			return salary*2;
		end;
end;

-- TYPE CREATED..

-- Query For the Table Creation:

create table emp2 of employee_t;
